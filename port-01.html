Drive Ownership Transfer & Scrub

Google Drive doesn't automatically transfer the contents of a folder when you transfer the parent. This script recursively transfers ownership of all nested files, and explicitly removes the previous owner's access.

Note: This script only works for accounts within the same Google Workspace domain. It must be run from the account that currently owns the files (or a Super Admin).

// How to Deploy

  1. Go to script.google.com and click New Project.
  2. Copy the code below and paste it into the editor, replacing any existing code.
  3. Update the folderId and target email addresses at the top of the script.
  4. Click the Save icon.
  5. Select transferFolderOwnership from the dropdown menu at the top and click Run. (You will be asked to authorize the script).
  6. Once the transfer is complete, select removeFolderAccess and run it to scrub the old user's permissions.

// The Code (Code.gs)

function transferFolderOwnership() {
  // --- CONFIGURATION ---
  var folderId = 'PASTE_YOUR_FOLDER_ID_HERE'; 
  var newOwnerEmail = 'correct.owner@yourdomain.com'; 
  // ---------------------

  try {
    var folder = DriveApp.getFolderById(folderId);
    transferOwnershipRecursively(folder, newOwnerEmail);
    folder.setOwner(newOwnerEmail);
    Logger.log('Successfully transferred main folder.');
  } catch (e) {
    Logger.log('Error accessing folder: ' + e.message);
  }
}

function transferOwnershipRecursively(folder, newOwnerEmail) {
  var files = folder.getFiles();
  while (files.hasNext()) {
    try { files.next().setOwner(newOwnerEmail); } catch (e) {}
  }
  var subfolders = folder.getFolders();
  while (subfolders.hasNext()) {
    var subfolder = subfolders.next();
    try {
      transferOwnershipRecursively(subfolder, newOwnerEmail);
      subfolder.setOwner(newOwnerEmail);
    } catch (e) {}
  }
}

function removeFolderAccess() {
  // --- CONFIGURATION ---
  var folderId = 'PASTE_YOUR_FOLDER_ID_HERE'; 
  var userToRemove = 'wrong.person@yourdomain.com'; 
  // ---------------------

  try {
    var folder = DriveApp.getFolderById(folderId);
    removeAccessFromItem(folder, userToRemove);
    removeAccessRecursively(folder, userToRemove);
    Logger.log('Successfully scrubbed access.');
  } catch (e) {
    Logger.log('Error accessing folder: ' + e.message);
  }
}

function removeAccessRecursively(folder, userEmail) {
  var files = folder.getFiles();
  while (files.hasNext()) { removeAccessFromItem(files.next(), userEmail); }
  var subfolders = folder.getFolders();
  while (subfolders.hasNext()) {
    var subfolder = subfolders.next();
    removeAccessFromItem(subfolder, userEmail);
    removeAccessRecursively(subfolder, userEmail);
  }
}

function removeAccessFromItem(item, userEmail) {
  try {
    var editors = item.getEditors();
    for (var i = 0; i < editors.length; i++) {
      if (editors[i].getEmail() === userEmail) item.removeEditor(userEmail);
    }
    var viewers = item.getViewers();
    for (var j = 0; j < viewers.length; j++) {
      if (viewers[j].getEmail() === userEmail) item.removeViewer(userEmail);
    }
  } catch (e) {}
}
