Drive Ownership Transfer & Scrub

Google Drive doesn't automatically transfer the contents of a folder when you transfer the parent. These two scripts handle the transfer and the cleanup without needing messy recursive helper functions.

REQUIREMENTS This script only works for accounts within the same Google Workspace domain. The scripts must be run by two separate users to fully hand off the data.

How to deploy this script

  1. The "Wrong" Person goes to script.google.com, pastes Script 1 into the editor, updates the variables, and clicks Run. This pushes ownership to the correct person.
  2. The "Correct" Person then goes to script.google.com, pastes Script 2 into the editor, updates the variables, and clicks Run. This permanently revokes the wrong person's remaining Editor/Viewer access.

Script 1: Push Ownership

Code.gs
function executeOwnershipTransfer() {
  // --- CONFIGURATION ---
  var folderId = 'PASTE_YOUR_FOLDER_ID_HERE'; 
  var correctOwnerEmail = 'correct.owner@yourdomain.com'; 
  // ---------------------

  try {
    var mainFolder = DriveApp.getFolderById(folderId);
    var foldersToProcess = [mainFolder]; 

    while (foldersToProcess.length > 0) {
      var currentFolder = foldersToProcess.shift();
      
      // 1. Transfer files in current folder
      var files = currentFolder.getFiles();
      while (files.hasNext()) {
        try { files.next().setOwner(correctOwnerEmail); } catch (e) {}
      }
      
      // 2. Queue up subfolders and transfer their ownership
      var subfolders = currentFolder.getFolders();
      while (subfolders.hasNext()) {
        var subfolder = subfolders.next();
        foldersToProcess.push(subfolder);
        try { subfolder.setOwner(correctOwnerEmail); } catch (e) {}
      }
    }
    
    // 3. Transfer the main parent folder last
    try { mainFolder.setOwner(correctOwnerEmail); } catch (e) {}
    
    Logger.log('Successfully transferred all ownership to: ' + correctOwnerEmail);
  } catch (e) {
    Logger.log('Critical Error: ' + e.message);
  }
}

Script 2: Scrub Previous Access

Code.gs
function executeAccessScrub() {
  // --- CONFIGURATION ---
  var folderId = 'PASTE_YOUR_FOLDER_ID_HERE'; 
  var wrongPersonEmail = 'wrong.person@yourdomain.com'; 
  // ---------------------

  try {
    var mainFolder = DriveApp.getFolderById(folderId);
    var foldersToProcess = [mainFolder];

    while (foldersToProcess.length > 0) {
      var currentFolder = foldersToProcess.shift();
      
      // 1. Scrub access from the folder itself
      try {
        var folderEditors = currentFolder.getEditors();
        for (var i = 0; i < folderEditors.length; i++) {
          if (folderEditors[i].getEmail() === wrongPersonEmail) currentFolder.removeEditor(wrongPersonEmail);
        }
        var folderViewers = currentFolder.getViewers();
        for (var j = 0; j < folderViewers.length; j++) {
          if (folderViewers[j].getEmail() === wrongPersonEmail) currentFolder.removeViewer(wrongPersonEmail);
        }
      } catch (e) {}

      // 2. Scrub access from files inside this folder
      var files = currentFolder.getFiles();
      while (files.hasNext()) {
        var file = files.next();
        try {
          var fileEditors = file.getEditors();
          for (var x = 0; x < fileEditors.length; x++) {
            if (fileEditors[x].getEmail() === wrongPersonEmail) file.removeEditor(wrongPersonEmail);
          }
          var fileViewers = file.getViewers();
          for (var y = 0; y < fileViewers.length; y++) {
            if (fileViewers[y].getEmail() === wrongPersonEmail) file.removeViewer(wrongPersonEmail);
          }
        } catch (e) {}
      }
      
      // 3. Queue up subfolders
      var subfolders = currentFolder.getFolders();
      while (subfolders.hasNext()) {
        foldersToProcess.push(subfolders.next());
      }
    }
    Logger.log('Successfully scrubbed all access for: ' + wrongPersonEmail);
  } catch (e) {
    Logger.log('Critical Error: ' + e.message);
  }
}
